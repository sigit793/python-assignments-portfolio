def overall(x):
    if x >= 0 and x <= 1:
        def error(c, d):
            y = (c**((2*d)+1))/((2*d)+1)
            return y

        j = 0
        while error(x, j) > 0.0001:
            j += 1

        approx = 0
        for i in range(j):
            value = (((-1)**i)*(x**((2*i)+1)))/((2*i)+1)
            approx += value

        return approx, j, error(x, j)
    else:
        return "error"

# List of test values
test_values = [-1, 0, 0.25, 0.5, 0.75, 1]

for value in test_values:
    result = overall(value)
    print(f"Input: {value}, Output: {result}")
